   def _round2(value):

   def calculate_tip(bill_total, tip_percentage, number_of_people):
       """Compute the tip, total-with-tip, and per-person split for a bill.
   
       Args:
           bill_total: The bill amount before tip (0.00 to 999,999,999.99).
           tip_percentage: The tip percentage (0 to 100).
           number_of_people: The number of people splitting the bill; an integer
               greater than or equal to 1.
   
       Returns:
           A dict with Decimal values under the keys ``tip_amount``,
           ``total_with_tip``, and ``per_person_amount``, each rounded to two
           decimal places using round-half-up.

       if isinstance(number_of_people, bool) or not isinstance(number_of_people, int):
           raise ValueError(
               "number of people must be an integer greater than or equal to 1"

       if number_of_people < 1:
           raise ValueError(
               "number of people must be an integer greater than or equal to 1"
           )

       pct = Decimal(str(tip_percentage))
